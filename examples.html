Quick Integration

Get started with NCS-API in minutes using your preferred technology stack 

// Install the NCS-API client
npm install @ncs-api/client

// Initialize and use
import NCSClient from '@ncs-api/client';

const client = new NCSClient('YOUR_API_KEY');

const result = await client.cluster({
  data: yourData,
  algorithm: 'ncs',
  clusters: 5
});

console.log('Clusters:', result.clusters);
console.log('Quality score:', result.quality);
# Install the NCS-API Python client
pip install ncs-api-client

# Initialize and use
from ncs_api import NCSClient

client = NCSClient(api_key='YOUR_API_KEY')

result = client.cluster(
    data=your_data,
    algorithm='ncs',
    clusters=5
)

print(f"Clusters: {result['clusters']}")
print(f"Quality score: {result['quality']}")
# Direct API call using cURL
curl -X POST https://api.ncs-cluster.com/v1/cluster \
  -H "Authorization: Bearer YOUR_API_KEY" \
  -H "Content-Type: application/json" \
  -d '{
    "data": [[1,2], [3,4], [5,6]],
    "algorithm": "ncs",
    "clusters": 2
  }'
// Node.js with Express integration
const express = require('express');
const NCSClient = require('@ncs-api/client');

const app = express();
const ncsClient = new NCSClient(process.env.NCS_API_KEY);

app.post('/analyze', async (req, res) => {
  try {
    const result = await ncsClient.cluster({
      data: req.body.data,
      algorithm: 'ncs',
      clusters: req.body.clusters || 'auto'
    });
    
    res.json(result);
  } catch (error) {
    res.status(500).json({ error: error.message });
  }
});
